- dp[i]表示前缀为i的区间最多能划分多少个互不重叠的完美区间，dp[i]在这个条件下是单调的。
- 显然，对于当前位置，要么不选，要么选择与它最近的区间去组合成新的完美区间。
- 注意可能有为0的情况，最好用count/contains
```cpp
map<int,int> mp;
int pre[N];
int a[N];
void Atomatic_AC_machine(){
    int n;cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
    for(int i=1;i<=n;i++){
        pre[i]=pre[i-1]+a[i];
    }
    vector<int> dp(n+1,0);
    mp[0]=0;
    for(int i=1;i<=n;i++){
        dp[i]=dp[i-1];
        if(mp.contains(pre[i])){
            dp[i]=max(dp[i],dp[mp[pre[i]]]+1);
        }
        mp[pre[i]]=i;
    }
    cout<<*max_element(dp.begin(),dp.end())<<endl;
    mp.clear();
}
```
